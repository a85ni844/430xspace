const ExpandableReply = ({ reply, index, replies, allPostProps, posts,
postsWithChildren }) => {
    const [isExpanded, setIsExpanded] = useState(false);
    const hasChildren = postsWithChildren.has(reply.id);

    const countDescendants = useCallback((postId) => {
        const children = posts.filter(p => p.replyToId === postId);
        let count = children.length;
        children.forEach(child => {
            count += countDescendants(child.id);
        });
        return count;
    }, [posts]);

    const descendantCount = useMemo(() => hasChildren ? countDescendants(reply.id) :
0, [hasChildren, reply.id, countDescendants]);

    return (
        <div>
            <PostItem
                {...allPostProps}
                post={reply}
                isThreadView={true} // 스레드 내 직계 댓글로 처리 (PostItem에서 border 결정)
                drawTopLine={true}
                drawBottomLine={hasChildren}
            />
            {hasChildren && !isExpanded && (
                <div className="pb-2" onClick={(e) => { e.stopPropagation();

setIsExpanded(true); }}>
                    <div className="relative pt-1 ml-16 cursor-pointer">
